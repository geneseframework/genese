                /**
                 * Get component data
                 */
                getData(): void {
                    this.functionalDamageDeletable = this.damageService.functionalDamageDeletable;
                    this.route.params.subscribe(params => {
                        this.damageId = params.damageId.toString();
                        if (!this.actionDone) {
                            if (this.damageService.versionNumber) {
                                this.versionNumber = this.damageService.versionNumber;
                                this.damageService.paramsUrl = `?version=${this.versionNumber}`;
                            }
                        } else {
                            this.damageService.paramsUrl = '';
                            if (this.versionToUpdate) {
                                this.damageService.versionNumber += 1;
                                this.versionNumber = this.damageService.versionNumber;
                                this.versionToUpdate = false;
                                this.damageService.isValidated = false;
                            }
                        }
                        this.spinnerService.startLoading();
                        let keepLoading = true;
                        this.functionalityId = params.functionalityId.toString();
                        this.functionalitiesGenese.getAll({gnPath: '/functionalities'})
                            .pipe(finalize(() => {
                                keepLoading ? keepLoading = false : this.spinnerService.stopLoading();
                            }))
                            .subscribe((functionalities: Functionalities[]) => {
                                const functionality = functionalities.find(e => e.id === this.functionalityId);
                                this.functionalityName = functionality ? functionality.name : '';
                            });
                        this.functionalityDamageGenese.getAll({
                            gnPath: `/damages/${this.damageId}/functionality/${this.functionalityId}${this.damageService.paramsUrl}`
                        })
                            .pipe(finalize(() => {
                                keepLoading ? keepLoading = false : this.spinnerService.stopLoading();
                            }))
                            .subscribe((fd: FunctionalityDamage[]) => {
                                this.functionalityDamageList = fd;
                            });
                        if (!this.damageService.isInit) {
                            this.damageService.behaviorUpdateDamage.next('getData');
                        } else {
                            this.damageService.isInit = false;
                        }
                    });
                }
