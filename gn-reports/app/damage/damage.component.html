                /**
                 * Archive given item
                 * @param item
                 */
                revokeDamage(item: SideListItem): void {
                    const revokeDamageDialog = this.dialog.open(DialogModalComponent, {
                        width: '600px',
                        height: '250px',
                        hasBackdrop: false,
                        data: {
                            title: {key: 'APP.DAMAGE_LIBRARY.DAMAGE_DELETE_MODAL.TITLE', params: {}},
                            body: {key: 'APP.DAMAGE_LIBRARY.DAMAGE_DELETE_MODAL.BODY', params: {damageName: item.name}},
                            buttonValidModal: {key: 'MODAL.DELETE', params: {}}
                        }
                    });
                    revokeDamageDialog.afterClosed().subscribe((data) => {
                        if (data) {
                            this.toasterService.displayToastDeleteSuccessResult(SuccessStatus.SUCCESS);
                            const itemVersion = item.versionNumber;
                            this.startLoading();
                            this.damagesGenese.deleteCustom(`/damages/${item.id}`)
                                .pipe(finalize(() => this.stopLoading()))
                                .subscribe(() => {
                                    this.damageListLoaded = false;
                                    if (this.selectedDamage) {
                                        if (itemVersion === '1' && (this.selectedDamage.id === item.id)) {
                                            this.selectedDamage = undefined;
                                            this.router.navigate(['damage']);
                                        } else if (this.selectedDamage.id === item.id) {
                                            this.damageService.refreshDamageVersion.next('refresh');
                                        }
                                    }
                                    this.getData();
                                });
                        }
                    });
                }

createDamage Cognitive complexity 19 Cyclomatic complexity 10 
                            
                                
            
            
                /**
                 * Create a new damage
                 */
                createDamage(): void {
                    this.translateService.get(['APP.DAMAGE_LIBRARY.NEW_DAMAGE']).subscribe(translations => {
                        this.damageList.push({id: '', name: translations['APP.DAMAGE_LIBRARY.NEW_DAMAGE']});
                    });
                    const createDamageDialog = this.dialog.open(DamageModalComponent, {
                        width: '800px',
                        height: '650px',
                        hasBackdrop: false,
                        data: {
                            mandatoryLanguages: [this.userLanguage]
                        }
                    });
                    createDamageDialog.afterClosed().subscribe((data) => {
                        if (data) {
                            this.startLoading();
                            this.damagesGenese
                                .createCustom('/damages', {translatedFields: data})
                                .pipe(finalize(() => this.stopLoading()))
                                .subscribe(async newDamage => {
                                    this.damageListLoaded = false;
                                    this.getData();
                                    const newDamageName = newDamage && newDamage.translatedFields && newDamage.translatedFields[this.userLanguage]
                                        ? newDamage.translatedFields[this.userLanguage].name
                                        : '';
                                    this.damageClicked({id: newDamage.id, name: newDamageName});
                                }, err => {
                                    console.error(err);
                                    this.damageList.pop();
                                });
                        } else {
                            if (data === '') {
                                this.damageList.pop();
                            }
                        }
                    });
                }
