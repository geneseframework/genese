                /**
                 * Returns selected options or placeholder value if nothing is selected.
                 * @returns {string}
                 */
                getSelectedOption(): string {
                    if (this.selectedOption && this.descriptor && this.descriptor.name) {
                        if (this.descriptor.value && this.selectOptions && this.selectOptions.length > 0) {
                            const selected = this.selectOptions.find((option) => option[this.descriptor.value] == this.selectedOption);
                            return selected ? this.getItemValue(selected, this.descriptor.name) : '';
                        } else {
                            return this.selectedOption[this.descriptor.name];
                        }
                    } else if (this.selectedOption) {
                        return this.selectedOption;
                    } else {
                        return this.placeholder;
                    }
                }

                getItemValue(item: any, givenPath: string) {
                    return givenPath.split('.').reduce((built, next) => {
                        return built ? built[next] : null;
                    }, item);
                }
