                /**
                 * Initialize fielLoader. This initialisation must be on ngOnChanges and not in ngOnInit,
                 * because the file is perhaps given in the parentComponent with an asynchronous process,
                 * like a translation
                 */
                ngOnChanges(changes: SimpleChanges) {
                    if (changes.files) {
                        if (this.files && Array.isArray(this.files)) {
                            this.fileList = [this.emptyFile];
                            this.files.forEach((file) => {
                                if (file.uuid && file.filename) {
                                    this.fileList.push({fileName: file.filename, isDocumentDownloadable: true, fileSize: '', uuid: file.uuid});
                                }
                            });
                        } else {
                            this.fileList = [this.emptyFile];
                        }
                    }
            
                    if (changes.fileExtension) {
                        this.parseFileExtension(this.fileExtension);
                    }
                }

                /**
                 * @param event
                 * @returns void
                 */
                onFileSelected(event): void {
                    const file: File = event.target.files[0];
                    if (file) {
                        this.fileLoading = true;
                        const fileDisplayTemp = {fileSize: '', fileName: file.name, isDocumentDownloadable: false, uuid: ''};
                        if (file.size) {
                            fileDisplayTemp.fileSize = file.size.toString();
                        }
            
                        if ((this.fileList[0] && this.fileList[0].fileName === '' && this.fileList[0].uuid === '') || !this.multi ) {
                            this.fileList[0] = fileDisplayTemp;
                        } else {
                            this.fileList.unshift(fileDisplayTemp);
            
                        }
            
                        this.fileService.preUpload(file)
                            .pipe(finalize(() => this.fileLoading = false))
                            .subscribe(
                            (fileData: { filename: string, uuid: string }) => {
                                if (fileData.uuid !== '') {
                                    this.fileSelected.emit({uuid: fileData.uuid, filename: fileData.filename});
                                    this.fileList[0].uuid = fileData.uuid;
                                    this.fileList[0].fileName = fileData.filename;
                                    this.fileList[0].fileSize = (event.target.files[0].size / 1000000).toFixed(2);
                                    this.fileList[0].isDocumentDownloadable = true;
                                    if (this.multi) {
                                        this.unshiftFirst();
                                    }
                                }
                            }, err => {
                            this.fileList[0] = this.emptyFile;
                        });
                    }
                }
