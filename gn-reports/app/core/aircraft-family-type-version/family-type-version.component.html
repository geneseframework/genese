                /**
                 * If the parent component changes its values (after receiving the result of an Observable), we refresh the values,
                 * but we do it only when the component is initialized the first time to avoid errors by emitting new values in the
                 * parent component (circular changes)
                 * @param changes
                 */
                ngOnChanges(changes: SimpleChanges): void {
                    if (this.isInitialization && (
                        (changes.selectedFamily && !changes.selectedFamily.firstChange)

                        this.isInitialization = false;

                    }
