                /**
                 * Check if the user has the access to the url in parameter
                 * @param url
                 */
                hasRights(url: string): boolean {
                    const features = JSON.parse(localStorage.getItem('features'));
                    let hasRights = false;
                    if (url !== undefined && Array.isArray(features)) {
                        hasRights = this.haveRightsForRoute(url.split('/')[0]);
                    }
                    if (!hasRights) {
                        /* istanbul ignore next */
                        this.translateService.get('TOAST.UNAUTHORIZED_ACCESS')
                            .subscribe(msg => this.toasterService.displayToast(msg, ColorPaletteTypes.warn, 3000));
                    }
                    return hasRights;
                }

                /**
                 * Check if for a given route, the user have rights or if the route doesn't exists
                 * @param route
                 */
                haveRightsForRoute(route: string): boolean {
                    let isKnownRoute = false;
                    const hasRights = Object.keys(RouteFeatures).some(key => {
                        if (key === route) {
                            isKnownRoute = true;
                        }
                        return (RouteFeatures[key] === route);
                    });
                    return hasRights || !isKnownRoute;
                }
