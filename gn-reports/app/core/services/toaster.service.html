                /**
                 * Display toast.
                 * @param {string} message - Message to display.
                 * @param {ColorPaletteTypes} type - Color palette type.
                 * @param {number} duration - Duration in milliseconds.
                 */
                /* istanbul ignore next */
                displayToast(message: string, type: ColorPaletteTypes, duration: number = 3000): void {
                    if (message) {
                        this.snackBar.openFromComponent(ToastComponent, {
                            duration,
                            data: message,
                            panelClass: `toast-${type}`,
                            horizontalPosition: 'center',
                            verticalPosition: 'top'
                        });
                    }
                }

                /**
                 * Display toast depending on success status.
                 * @param message
                 * @param successStatus
                 */
                /* istanbul ignore next */
                displayToastSuccessStatus(message: string, successStatus: SuccessStatus): void {
                    if (successStatus) {
                        const color: ColorPaletteTypes = successStatus === SuccessStatus.SUCCESS ? ColorPaletteTypes.primary : ColorPaletteTypes.warn;
                        this.displayToast(message, color);
                    }
                }

                /**
                 * Display toast for save operations
                 * @param successStatus
                 */
                /* istanbul ignore next */
                displayToastSaveSuccess(successStatus: SuccessStatus): void {
                    if (successStatus) {
                        const message: string = successStatus === SuccessStatus.SUCCESS
                            ? this.translateService.instant('TOAST.SAVED_SUCCESSFULLY')
                            : this.translateService.instant('TOAST.SAVE_FAILED');
                        this.displayToastSuccessStatus(message, successStatus);
                    }
                }

                /**
                 * Display toast for delete operations
                 * @param successStatus
                 */
                /* istanbul ignore next */
                displayToastDeleteSuccessResult(successStatus: SuccessStatus): void {
                    if (successStatus) {
                        const key: string = successStatus === SuccessStatus.SUCCESS ? 'TOAST.DELETED_SUCCESSFULLY' : 'TOAST.DELETE_FAILED';
                        this.translateService.get(key).subscribe(message => {
                            this.displayToastSuccessStatus(message, successStatus);
                        });
                    }
                }
