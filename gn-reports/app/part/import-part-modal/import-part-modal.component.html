                /**
                 * Get the parts
                 */
                getParts(): void {
                    this.startLoading();
                    this.partGenese
                        .getAllWithPagination('/parts', {
                            filters: {status: 'VALIDATED'},
                            pageSize: 20,
                            pageIndex: 0
                        })
                        .pipe(finalize(() => this.stopLoading()))
                        .subscribe((partsData: GetAllResponse<Part>) => {
                            this.pnOrPnRootSelectOptions = [];
                            if (partsData && partsData.results) {
                                partsData.results.forEach(item => {
                                    if (item.id.toString() !== this.data) {
                                        if (item.status === 'VALIDATED') {
                                            if (item.partNumber
                                                && !this.pnOrPnRootSelectOptions.find(e => e.content === item.partNumber)) {
                                                this.pnOrPnRootSelectOptions.push({value: item, content: item.partNumber});
                                            }
                                            if (item.partNumberRoot
                                                && !this.pnOrPnRootSelectOptions.find(e => e.content === item.partNumberRoot)) {
                                                this.pnOrPnRootSelectOptions.push({value: item, content: item.partNumberRoot});
                                            }
                                        }
                                    }
                                });
                            }
                            this.pnOrPnRootSelectOptionsBackup = [...this.pnOrPnRootSelectOptions];
                        });
                }
