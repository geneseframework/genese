                /**

                    const searchQuery = searchText ? `?search=${searchText}` : '';

                                    return {
                                        content: e,
                                        value: e
                                    };
                                });
                            }
                            return selectOptions;
                        })
                    );
                }

                /**
                 * Get mapping
                 * @param part
                 */
                getMapping(part: any) {
                    const url = `${environment.urls.baseApiUrl}${this.apiPath}/${part.id}${part.version ? '?version=' + part.version : ''}`;
                    return this.http.get(url).pipe(
                        map((data: any) => {
                            const functionalAreas: Array<FunctionalArea> = [];
                            if (data && Array.isArray(data.functionalAreas)) {
                                data.functionalAreas.forEach((fa) => {
                                    functionalAreas.push(FunctionalArea.mapToFunctionalArea(fa));
                                });
                            }
                            return functionalAreas;
                        })
                    );
                }

                /**
                 * Save the given mapping
                 * @param partId: number
                 * @param functionalAreas: Array<FunctionalArea>
                 * @param publishing: boolean
                 */
                saveMapping(partId: string, functionalAreas: FunctionalArea[], publishing?: boolean) {
                    let url = `${environment.urls.baseApiUrl}${this.apiPath}/${partId}${this.mappingUrl}`;
                    url += `${publishing ? `?validate=${publishing}` : ''}`;
                    const body = [];
                    functionalAreas.forEach((area: FunctionalArea) => {
                        body.push({
                            id: area.id,
                            areaNumber: area.areaNumber,
                            classification: area.classification,
                            disabled: area.disabled,
                            faNameId: area.functionalAreaName ? area.functionalAreaName.id : undefined,
                            functionalityId: area.functionality ? area.functionality.id : undefined,
                            material: area.material,
                            treatmentId: area.treatment ? area.treatment.id : undefined
                        });
                    });
                    return this.http.post(url, body);
                }
