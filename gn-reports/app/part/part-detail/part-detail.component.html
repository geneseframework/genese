                /**
                 * Check if the new number is valid and exchange it with another area if already taken
                 * @param partArea
                 * @param event
                 */
                onNumberChange(partArea: FunctionalArea, event: any): void {
                    const value = event.target.value;
                    if (value > 0 && (value !== partArea.areaNumber)) {
                        const already = this.part.functionalAreas.find((area: FunctionalArea) => area.areaNumber === value);
                        if (already) {
                            already.areaNumber = partArea.areaNumber;
                        }
                        partArea.areaNumber = value;
                        this.appService.dataChanged = true;
                        this.onFunctionChange();
                    }
                }

                functionalAreasValid(): boolean {
