                /**
                 * Map body to update
                 * @param childRequest
                 * @param status
                 * @param isCreation
                 * @return any
                 */
                /* istanbul ignore next */
                mapToUpdate(childRequest: ChildRequest, status: string, isCreation: boolean): any {
                    if (!isCreation && (!childRequest || !childRequest.id)) {
                        return of(undefined);
                    }
                    const clientsIds = [];
                    childRequest.childRequestClients.forEach((client) => {
                        clientsIds.push(+client.id);
                    });
                    return {
                        aircraftFamilyId: childRequest.aircraftFamily ? parseInt(childRequest.aircraftFamily.id, 10) : '',
                        aircraftTypeIds: Array.isArray(childRequest.aircraftTypes) && !!childRequest.aircraftTypes.length ?
                            childRequest.aircraftTypes.map(e => +e.id) : [],
                        aircraftVersionIds: Array.isArray(childRequest.aircraftVersions) && !!childRequest.aircraftVersions.length ?
                            childRequest.aircraftVersions.map(e => +e.id) : [],
                        clientIds: clientsIds ? clientsIds : [],
                        drtToInspect: childRequest.drtTotal ? +childRequest.drtTotal : 0,
                        environmentId: childRequest.environment ? parseInt(childRequest.environment.id, 10) : '',
                        isDeletable: childRequest.isDeletable ? childRequest.isDeletable : true,
                        medias: Array.isArray(childRequest.medias) && childRequest.medias.length > 0
                            ? childRequest.medias.map(m => m.uuid) : [],
                        missionTypeId: childRequest.missionType ? parseInt(childRequest.missionType.id, 10) : '',
                        partId: childRequest.partId ? parseInt(childRequest.partId, 10) : '',
                        serialNumbers: childRequest.serialNumbers ? childRequest.serialNumbers : [],
                        status
                    };
                }
