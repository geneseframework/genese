                /**
                 * Map to HeaderData
                 * @param request
                 * @param status
                 * @return HeaderData
                 */
                /* istanbul ignore next */
                mapToHeader(request: Request, status?: string, childRequest?: ChildRequest): HeaderData {
                    const headerData: HeaderData = {};
                    if (request) {
                        headerData.drt = childRequest && childRequest.drtTotal ? childRequest.drtTotal : '';
                        headerData.id = request.id ? request.id : '';
                        headerData.name = request.name ? request.name : '';
                        headerData.description = request.description ? request.description : '';
                        headerData.origin = request.origin ? request.origin.name : '';
                        headerData.airbusEntity = request.airbusEntity ? request.airbusEntity.countryName : '';
                        headerData.creationDate = request.creationDate
                            ? moment(request.creationDate).format('YYYY-MM-DD') : '';
                        headerData.dueDate = request.dueDate ? request.dueDate : '';
            
                        if (childRequest && childRequest.versionNumber) {
                            headerData.version = childRequest.versionNumber;
                        } else if (request.versionNumber) {
                            headerData.version = request.versionNumber;
                        } else {
                            headerData.version = '';
                        }
            
                        headerData.validator = request.validator ?
                            UtilsService.capitalize(request.validator.firstName) + ' ' + UtilsService.capitalize(request.validator.lastName) : '';
                        headerData.requester = request.requester ?
                            UtilsService.capitalize(request.requester.firstName) + ' ' + UtilsService.capitalize(request.requester.lastName) : '';
                        headerData.status =  request.status ? 'STATUS.' + request.status : '';
                        if (status) {
                            headerData.status = status;
                        } else if (request.status) {
                            headerData.status = request.status;
                        }
                    }
                    return headerData;
                }

                /**
                 * Change format of string date
                 * @param date: string
                 */
                /* istanbul ignore next */
                convertDate(date: string) {
                    if (date) {
                        const dates = date.split('-');
                        return dates[2].substring(0, 2) + '/' + dates[1] + '/' + dates[0];
                    }
                    return '';
                }
