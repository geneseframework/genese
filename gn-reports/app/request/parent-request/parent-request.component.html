                /**
                 * Check if parent request details is valid
                 */
                isValidRequest(fromSave?: boolean): boolean {
                    let dataChanged = false;
                    if (fromSave) {
                        if (this.request.status !== Status.VALIDATED) {
                            dataChanged = this.appService.dataChanged;
                        } else {
                            // User mustn't save a validated request
                            return false;
                        }
                    } else {
                        if (this.appService.dataChanged) {
                            dataChanged = true;
                        } else {
                            dataChanged = this.request.status !== Status.VALIDATED;
                        }
                    }
            
                    return (!!this.request.origin && this.request.origin.id !== '')
                        && !!this.request.reference
                        && !!this.request.aircraftFamily && !!this.request.aircraftTypes && !!this.request.aircraftVersions
                        && !!this.request.aircraftTypes && !!this.request.aircraftVersions
                        && !!this.request.ataCode
                        && Array.isArray(this.request.clients)
                        && this.request.clients.length > 0
                        && Array.isArray(this.request.operators)
                        && this.request.operators.length > 0
                        && Array.isArray(this.request.technicalResponsibles)
                        && this.request.technicalResponsibles.length > 0
                        && !!this.request.oetp
                        && dataChanged;
                }
