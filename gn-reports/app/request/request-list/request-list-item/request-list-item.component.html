loadChildData Cognitive complexity 5 Cyclomatic complexity 5 
                            
                                
            
                /**
                 * api call for child data & manager
                 * @param pageIndex
                 */
                loadChildData(pageIndex) {
                    this.pageIndex = pageIndex;
                    this.childRequestGenese.getAll({
                        gnPath: '/requests/' + this.item.id + '/child-requests?',
                        gnPage: this.pageIndex,
                        gnLimit: this.pageSize
                    })
                        .subscribe((childList: any) => {
                            if (childList && childList.results) {
                                this.childList = childList.results;
                                this.listLength = childList.totalResults;
                            }
                        });
            
                    this.technicalManagerGenese.getAll({
                        gnPath: '/requests/' + this.item.id + '/managers'
                    })
                        .subscribe(
                            (managers: String[]) => {
                                this.technicalManagers = managers;
                            });
                }
