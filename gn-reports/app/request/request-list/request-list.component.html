                /**
                 * Load all the request list data
                 */
                getRequestListData(isSort?: boolean, isFilter?: boolean): void {
                    if (isSort) {
                        let sortValue = '';
                        if (this.sortRequest.orderBy) {
                            sortValue += '&sortBy=' + this.sortRequest.orderBy;
                        }
                        if (this.paginator.pageIndex) {
                            sortValue += '&page=' + this.paginator.pageIndex;
                        }
                        if (this.paginator.pageSize) {
                            sortValue += '&size=' + this.paginator.pageSize;
                        }
                        if (this.sortRequest.sortDirection) {
                            sortValue += '&sortDirection=' + this.sortRequest.sortDirection;
                        }
                        this.sortValue = sortValue.substr(1);
                    }
            
                    if (isFilter) {
                        let filterValue = '';
                        if (this.filterRequest.name) {
                            filterValue += '&name=' + this.filterRequest.name;
                        }
                        if (this.filterRequest.entitys) {
                            for (const entity of this.filterRequest.entitys) {
                                filterValue += '&airbusEntityIds=' + entity;
                            }
                        }
                        if (this.filterRequest.requesters) {
                            for (const requester of this.filterRequest.requesters) {
                                filterValue += '&requesterIds=' + requester;
                            }
                        }
                        if (this.filterRequest.origins) {
                            for (const origin of this.filterRequest.origins) {
                                filterValue += '&originIds=' + origin;
                            }
                        }
                        if (this.filterRequest.status) {
                            for (const statusValue of this.filterRequest.status) {
                                filterValue += '&statuses=' + statusValue.toLocaleUpperCase();
                            }
                        }
                        this.filterValue = filterValue;
                    }
            
                    this.lightRequestGenese.getAll({
                        gnPath: '/requests?' + this.sortValue + this.filterValue
                    })
                        .subscribe((data: any) => {
                            this.requestList = data && Array.isArray(data.results) ? data.results : [];
                            for (const lightRequest of this.requestList) {
                                lightRequest.creationDate = this.requestService.convertDate(lightRequest.creationDate);
                            }
                            this.paginator.length = data && typeof data.totalResults === 'number' ? data.totalResults : 0;
                            this.emptyList = data.totalResults === 0;
                        });
                }
