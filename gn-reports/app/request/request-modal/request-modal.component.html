                /**
                 * return valid if the form tab is valid and invalid if not
                 * @return boolean
                 */
                isFormTabValid(language: Language): FormValidityType {
                    let isValid = this.form && this.form[language]
                    && this.form[language].name
                    && this.airbusEntity
                    && this.airbusEntity.id
                    && this.dueDateSelected
                        ? FormValidityType.valid
                        : FormValidityType.invalid;
                    if (
                            (language === Language.en
                            && !this.form[language].description
                            && this.form[this.userLanguage].description)
                            ||
                            (language === this.userLanguage
                                && !this.form[language].description
                                && this.form[Language.en].description)
                        ) {
                        isValid = FormValidityType.invalid;
                    }
                    return isValid;
                }
