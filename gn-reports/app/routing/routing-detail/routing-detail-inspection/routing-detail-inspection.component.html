                /**
                 * Get routing detail

                getDetail() {
                    if (this.operation && this.operation.id && this.task && this.task.id) {
                        this.startLoading();
                        const request: Observable<any>[] = [
                            this.routingOperationDataService.getRoutingInspectionDetail(this.routingId,
                                this.operation.id, this.task.id, this.versionNumber)
                        ];
                        this.operation && this.operation.behavior !== OperationBehavior.TODO_LIST ? request.push(
                            this.routingOperationDataService.getHeaderInspectionDetail(this.routingId, this.operation.id,
                                this.task.id, this.versionNumber)
                        ) : noop();
                        forkJoin(request).pipe(finalize(() => this.stopLoading()))
                            .subscribe(([inspectionData, headerData]: any) => {
                                    let hasInspections = false;
                                    let hasNewInspections = false;
                                    this.routingComponentMissing = false;
                                    this.newInspections = [];
                                    this.inspectionData = [];
                                    this.headerData = {};
                                    headerData ? this.headerData = headerData : this.headerData.functionalAreaName = this.task.name;
                                    this.functionalAreaHeaderDisplay = this.operation.template !== 'preliminary' &&
                                        this.operation.template !== 'closure';
            
                                    if (inspectionData && inspectionData.newInspections && inspectionData.newInspections.length > 0
                                        && this.hasWriteLevel) {
                                        hasNewInspections = true;
                                        this.newInspections = inspectionData.newInspections;
                                    }
                                    if (inspectionData && inspectionData.inspections && inspectionData.inspections.length > 0) {
                                        hasInspections = true;
                                        this.inspectionData = inspectionData.inspections;
                                        this.computeHiddenSteps();
                                        this.routingComponentMissing = false;
                                        this.inspectionsBackup = UtilsService.deepCopy(inspectionData.inspections);
                                    }
                                    if (!hasInspections && !hasNewInspections) {
                                        this.routingComponentMissing = true;
                                        this.inspectionData = [];
                                        this.hiddenSteps = [];

                                    }

computeHiddenSteps Cognitive complexity 10 Cyclomatic complexity 5 
                            
                                
            
            
                /**
                 * compute witch step is hidden
                 */
                computeHiddenSteps() {
                    this.inspectionData.forEach((inspection, inspectionIndex) => {
                        if (inspection.steps) {
                            inspection.steps.forEach((step, stepIndex) => {
                                if (!this.hiddenSteps[inspectionIndex]) {
                                    this.hiddenSteps[inspectionIndex] = [];
                                }
                                this.hiddenSteps[inspectionIndex][stepIndex] = !step.activated;
                            });
                        }
                    });
                }
