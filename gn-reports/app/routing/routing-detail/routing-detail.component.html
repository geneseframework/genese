                operationRemoved(wrapper: { operation: RoutingOperation, index: number }) {

                        const index = this.operations.findIndex((op: RoutingOperation) => wrapper.operation.id === op.id);
                        if (index > -1) {
                            this.removedOperations.push(wrapper.operation);
                            this.appService.dataChanged = true;
                            this.operations.splice(index, 1);

                        this.operations.splice(wrapper.index, 1);
                    }
                    this.operations.sort(this.routingService.sortByOrderNumber);
                    this.operations.forEach((operation, index) => {
                        if (index === 0 && operation.orderNumber !== 1) {
                            operation.orderNumber = 1;
                        } else if (index && ((operation.orderNumber - this.operations[index - 1].orderNumber) > 1)) {
                            operation.orderNumber = this.operations[index - 1].orderNumber + 1;
                        }
                    });
                    // todo rework internal and external so you can remove this.
                    this.operations = [...this.operations];
                    this.selectedTaskChanged = true;
                }

                operationOrderChanged(wrapper: { operation: RoutingOperation, index: number, increment: number }) {

                        const newOrder = oldOrder + wrapper.increment;

                /**
                 * add new Operation

                addOperation() {
                    if (this.newOperation) {
                        let orderNumber = 1;
                        this.operations.forEach((op) => {
                            if (!orderNumber || op.orderNumber >= orderNumber) {
                                orderNumber = op.orderNumber + 1;
                            }
                        });
                        const operation: RoutingOperation = {
                            operationTypeId: this.newOperation.operationType.id,
                            name: this.newOperation.operationType.name,
                            template: this.newOperation.operationType.template,
                            behavior: this.newOperation.operationType.behavior,
                            routingId: +this.routingId,
                            orderNumber,
                            inspectionValue: 'internal'
                        };
                        if (this.newOperation.tasks && this.newOperation.tasks.length > 0) {
                            operation.todoLists = this.newOperation.tasks;
                        }
                        this.operations = [...this.operations, operation];
                        this.newOperation = {
                            operationType: undefined,
                            tasks: []
                        };
                        this.appService.dataChanged = true;
                        this.selectedTaskChanged = true;
                    }
                }
