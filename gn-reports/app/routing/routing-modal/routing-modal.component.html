                /**
                 * Modal validation
                 */
                validModal(): void {
                    this.startLoading();
                    const translatedFields = {};
                    Object.keys(this.form).forEach((lang) => {
                        translatedFields[lang] = {
                            name: this.form[lang].name
                        };
                    });
                    const routing: CreateRouting = {
                        partId: this.partId ? this.partId : undefined,
                        partNumberRoot: this.partNumberRoot ? this.partNumberRoot : '',
                        translatedFields
                    };
            
                    if (!this.idRoutingToDuplicate) {
                        this.routingCreateResponseGenese.createCustom('/routings', routing)
                            .pipe(finalize(() => this.stopLoading()))
                            .subscribe(
                                (data: RoutingCreateResponse) => this.beforeClosingModal(data),
                                err => console.error(err)
                            );
                    } else {
                        this.routingCreateResponseGenese
                            .createCustom(`/routings/${this.idRoutingToDuplicate}/duplicate`, routing)
                            .pipe(finalize(() => this.stopLoading()))
                            .subscribe(
                                (data: any) => this.beforeClosingModal(data),
                                err => console.error(err)
                            );
                    }
                }

                /**
                 * return valid if the form tab is valid and invalid if not
                 * @return boolean
                 */
                isFormTabValid(language: Language): FormValidityType {
                    return this.form && this.form[language]
                    && (this.partId || this.partNumberRoot)
                    && !!this.form[language].name
                        ? FormValidityType.valid
                        : FormValidityType.invalid;
                }
