                /**
                 * Navigate to other pages if no data changed
                 * @param paths
                 * Ex: ['damage', `${damageId}`, `${functionalityId}`]
                 */
                /* istanbul ignore next */
                navigateIfNoChanges(paths: string[]): void {
                    if (this.dataChanged) {
                        this.checkUpdates().subscribe((data: boolean) => {
                            if (data) {
                                this.dataChanged = false;
                                this.router.navigate(paths);
                            }
                        });
                    } else {
                        this.router.navigate(paths);
                    }
                }
