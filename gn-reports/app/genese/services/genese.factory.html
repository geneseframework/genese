                    if (data) {
                        return this.gMapperFactory.mapToObject<U>(data, uConstructor);

getOne Cognitive complexity 10 Cyclomatic complexity 7 
                            
                                
            
            
            
                /**
                 * Get one element of the T class
                 * If id parameter is undefined, the request just calls http GET with the path param
                 * If id is defined, the request calls http GET with the path given by the apiRoot method
                 * @param {string} id
                 * @param uConstructor
                 */
                /* istanbul ignore next */
                getOne(id?: string, path?: string, params?: { [key: string]: string }): Observable<T> {
                    if (!id && !path) {
                        console.error('No id : impossible to get element');
                        return of(undefined);
                    }
                    // FOR MODELISATOR ; Changement à valider
                    const url = id ? this.apiRoot(path) + '/' + id : path;
                    let httpParams = new HttpParams();
                    if (params) {
                        for (const key of Object.keys(params)) {
                            if (params[key]) {
                                httpParams = httpParams.set(key, params[key]);
                            }
                        }
                    }
                    const options = {params: httpParams};
                    return this.http.get(url, options)
                        .pipe(
                            map((data: any) => {
                                return this.mapToObject<T>(data);
                            })
                        );
                }

getAll Cognitive complexity 21 Cyclomatic complexity 10 
                            
                                
            
                /**
                 * Get all elements
                 * If the http response format have this format :
                 * {
                 *      totalResults?: number;
                 *      totalPages?: number;
                 *      results: any;
                 * }
                 *
                 * If not, it returns T[] object
                 * @param uConstructor
                 * @param params
                 */
                /* istanbul ignore next */
                getAll<U = T>(params: GnRequest): Observable<GetAllResponse<U> | U[]> {
                    if (!params || !params.gnPath) {
                        console.error('Path undefined in getAll()');
                        return of(undefined);
                    }
                    let httpParams = new HttpParams();
                    httpParams = params.gnPage !== undefined ? httpParams.set('page', params.gnPage.toString()) : httpParams;
                    httpParams = params.gnLimit ? httpParams.set('size', params.gnLimit.toString()) : httpParams;
                    httpParams = params.gnExtract ? httpParams.set('gnExtract', JSON.stringify(params.gnExtract)) : httpParams;
                    if (params.gnFilters) {
                        for (const key of Object.keys(params.gnFilters)) {
                            if (params.gnFilters[key]) {
                                httpParams = httpParams.set(key, params.gnFilters[key].toString());
                            }
                        }
                    }
                    const options = {params: httpParams};
                    const url = params && params.gnPath ? this.geneseEnvironment.api + params.gnPath : this.apiRoot();
                    return this.http.get(url, options).pipe(
                        map((response: any) => {
                            if (response) {
                                if (this._responseWithPagination(response)) {
                                    return {
                                        results: this.gMapperFactory.mapGetAllResults(response.results),
                                        totalResults: response.totalResults};
                                } else {
                                    return this.gMapperFactory.mapGetAllResults(response);
                                }
                            } else {
                                return [];
                            }
                        })
                    );
                }

                delete(id?: string, path?: string): Observable<ResponseStatus> {
                    if (!id && !path) {
                        console.error('%c Error deleting element: undefined id and url ');
                        return of(undefined);
                    } else {
                        const url = path ? this.geneseEnvironment.api + path : this.apiRoot();
                        return this.http.delete(url + '/' + id, {observe: 'response'})
                            .pipe(
                                map((response: HttpResponse<any>) => {
                                    return response && response.ok === true ? ResponseStatus.SUCCESS : ResponseStatus.FAILED;
                                })
                            );
                    }
                }
