                /**
                 * Modal validation
                 */
                validModal(): void {
                    this.startLoading();
                    if (this.form.equipmentNumber !== '') {
                        this.equipmentNumber = this.form.equipmentNumber.toUpperCase();
                    }
                    if (this.form.sn !== '') {
                        this.serialNumber = this.form.sn.toUpperCase();
                    }
            
                    const bodyFiltering = {
                        equipmentNumber: this.equipmentNumber,
                        partNumber: this.partNumberDisplayed.toUpperCase(),
                        serialNumber: this.serialNumber
                    };
            
                    this.filteringGenese.create(bodyFiltering, '/filterings')
                        .pipe(finalize(() => this.stopLoading()))
                        .subscribe((filteringCreated) => {
                            this.removeOverlay();
                                this.dialogRef.close(true);
                                this.router.navigate(['filtering', filteringCreated.id, 'edit']);
                            },
                            err => console.error(err)
                        );
                }
