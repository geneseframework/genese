buildQueryParams Cognitive complexity 3 Cyclomatic complexity 3 
                            
                                
            
            
                /**
                 * build url suffix with given params
                 */
                buildQueryParams(paramWrapper: { [key: string]: any }) {
                    let queries = '';
                    Object.keys(paramWrapper).forEach((key: string) => {
                        if (paramWrapper[key]) {
                            queries = `${queries}&${key}=${paramWrapper[key]}`;
                        }
                    });
            
                    return queries.replace('&', '?');
                }

                getRoutingComponents(pageNumber = 0,
                                     pageSize = 20,
                                     filters?: RoutingComponentListFilter) {
            
                    const queryParams = this.buildQueryParams({
                        page: pageNumber.toString(),
                        size: pageSize.toString(),
                        operationTypeId: filters.operationTypeId ? filters.operationTypeId.toString() : undefined,
                        functionalityId: filters.functionalityId ? filters.functionalityId.toString() : undefined,
                        todoListNameId: filters.todoListNameId ? filters.todoListNameId.toString() : undefined,
                        subTaskId: filters.subTaskId ? filters.subTaskId.toString() : undefined,
                        inspectionId: filters.inspectionId ? filters.inspectionId.toString() : undefined,
                        status: filters.status ? filters.status.toString() : undefined
                    });
            
                    const url = `${environment.urls.baseApiUrl}${this.routingUrl}${queryParams}`;
                    return this.http.get(url).pipe(
                        map((data: any) => {
                            const components = [];
                            if (data.results && data.results.length > 0) {
                                data.results.forEach((component) => {
                                    components.push(RoutingComponentList.mapToComponentList(component));
                                });
                            }
                            return {components, totalResults: data.totalResults, totalPages: data.totalPages};
                        }));
                }
