getImageUrl Cognitive complexity 9 Cyclomatic complexity 5 
                            
                                
            
            
                /**
                 * Compute image url
                 * @param uuid
                 * @param width
                 * @param height
                 */
                getImageUrl(uuid: string, width?: number, height?: number): string {
                    let paramsUrl = '';
                    if (width && height) {
                        paramsUrl = `?height=${height}&width=${width}`;
                    } else if (!!width) {
                        paramsUrl = `?width=${width}`;
                    } else if (!!height) {
                        paramsUrl = `?height=${height}`;
                    }
                    return `${environment.urls.baseApiUrl}/api/media/${uuid}${paramsUrl}`;
                }
