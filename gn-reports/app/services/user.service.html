getUsers Cognitive complexity 10 Cyclomatic complexity 5 
                            
                                
            
            
                /**
                 * Get the users
                 * @param onlyActive
                 * @param filter
                 * @param airbusEntityId
                 * @param pageNumber
                 * @param pageSize
                 */
                /* istanbul ignore next */
                getUsers(onlyActive: boolean, filter = '', airbusEntityId: number, pageNumber = 0, pageSize = 20) {
                    return this.httpClient.get(this.apiBaseUrl + this.apiPath, {
                        params: new HttpParams()
                            .set('onlyActive', onlyActive.toString())
                            .set('airbusEntityId', airbusEntityId.toString())
                            .set('page', pageNumber.toString())
                            .set('size', pageSize.toString())
                            .set('search', filter.toString())
                    }).pipe(
                        map((data: any) => {
                            const users: User[] = [];
                            for (const user of data.results) {
                                const userRoles = user.roles;
                                user.roles = [];
                                if (userRoles.length > 0) {
                                    for (const role of userRoles) {
                                        user.roles.push(Role.mapToRole(role));
                                    }
                                }
                                users.push(User.mapToUser(user));
                            }
                            return {
                                totalResults: data.totalResults,
                                totalPages: data.totalPages,
                                users
                            };
                        })
                    );
                }

                /**
                 * Get user features
                 * @param user
                 */
                getUserFeatures(user: User): Feature[] {
                    const userFeatures = [];
                    if (user.features) {
                        user.features.forEach(feature => {
                            userFeatures.push({
                                code: feature.code,
                                rightLevel: feature.rightLevel
                            });
                        });
                    }
                    return userFeatures;
                }

                /**
                 * Check if the user has the write level to the resource
                 * @param featureCode: ('PART_MAPPING', 'DASHBOARD', 'ROUTING', 'DAMAGE', 'FILTERING', 'ADMIN', 'REQUEST', 'DRT_INTERNAL_INSPECTION',
                 * ROUTING_COMPONENT')

                /* istanbul ignore next */
                hasWriteLevel(featureCode: string): boolean {
                    const features = JSON.parse(localStorage.getItem('features'));
                    let hasRights = false;
                    if (Array.isArray(features)) {
                        hasRights = features.some(feature => {
                            return feature.code === featureCode && feature.rightLevel === RightLevel.WRITE;
                        });
                    }
                    return hasRights;
