                /**
                 * Routing compoenent mapper
                 * @param data
                 */
                mapToRoutingComponent(data: any): RoutingComponent {
                    const routingComponent = new RoutingComponent();
            
                    let operationType;
                    if (data && data.operationType && data.operationType.id && data.operationType.template) {
                        operationType = {
                            id: `${data.operationType.id}`,
                            name: data.operationType.name,
                            template: data.operationType.template,
                            behavior: data.operationType.behavior
                        };
                    }
            
                    if (data) {
                        routingComponent.operationType = operationType;
                        routingComponent.taskId = data.taskId ? `${data.taskId}` : undefined;
                        routingComponent.subTaskId = data.subTaskId ? `${data.subTaskId}` : undefined;
                        routingComponent.inspectionValue = data.inspectionValue ? data.inspectionValue : 'generic';
                        routingComponent.creationDate = data.creationDate ? data.creationDate : '';
                        routingComponent.status = data.status ? data.status : '';
                        routingComponent.id = data.routingComponentIndexId ? data.routingComponentIndexId : undefined;
                        routingComponent.versionNumber = data.versionNumber;
                        routingComponent.isLatestVersion = data.isLatestVersion;
                        if (data.steps) {
                            const steps = [];
                            data.steps.forEach(step => steps.push(Step.mapToStep(step)));
                            routingComponent.steps = steps;
                        } else {
                            routingComponent.steps = undefined;
                        }
                    }
            
                    return routingComponent;
                }

                /**
                 * Retrieve component
                 * @param id
                 * @param params
                 */
                getComponent(id?: string, version?: string): Observable<RoutingComponent> {
                    if (!id) {
                        return of(undefined);
                    }
                    let url = `${environment.urls.baseApiUrl}${this.componentUrl}/${id}`;
                    if (version) {
                        url += `?version=${version}`;
                    }
                    return this.http.get(url).pipe(
                        map((data: any) => this.mapToRoutingComponent(data))
                    );
                }

                /**
                 * Edit an existing component
                 * @param component
                 * @param steps
                 * @param validate
                 */
                setComponent(component: RoutingComponent, steps?: Step[], validate?: boolean): Observable<RoutingComponent> {
                    const body = {
                        id: component.id ? component.id : undefined,
                        inspectionValue: component.inspectionValue,
                        operationTypeId: component.operationType ? component.operationType.id : undefined,
                        steps: steps ? steps.map((item) => {
                            if (item.attachedFiles) {
                                item.mediaUuids = item.attachedFiles.map(x => x.uuid);
                                delete item.attachedFiles;
                            }
                            return item;
                        }) : [],
                        subTaskId: component.subTaskId,
                        taskId: component.taskId
                    };
            
                    if (component.id) {
                        const url =
                            `${environment.urls.baseApiUrl}${this.componentUrl}/${component.id}${validate ? `?validate=${validate}` : ''}`;
                        return this.http.put(url, body).pipe(map((data: any) => this.mapToRoutingComponent(data)));
                    } else {
                        const url = `${environment.urls.baseApiUrl}${this.componentUrl}${validate ? `?validate=${validate}` : ''}`;
                        return this.http.post(url, body).pipe(map((data: any) => this.mapToRoutingComponent(data)));
                    }
                }

                /**
                 * Upload new file
                 * @param file
                 */
                uploadFile(file: File) {
                    const url = `${environment.urls.baseApiUrl}/api/media/pre-upload`;
                    return this.fileService.postFile(file, url).pipe(
                        map((data: any) => {
                            let extension = '';
                            if (data.filename) {
                                const fileNameSplit = data.filename.split('.');
                                extension = fileNameSplit.pop();
                            }
                            return {
                                extension,
                                filename: data.filename,
                                uuid: data.uuid
                            };
                        })
                    );
                }
