                /**
                 * Fill all posts and thingworx inspection picture from current automatic step
                 * @param step
                 * @param filledPost
                 * @param inspectionDataIndex
                 * @param entryIndex
                 * @param entry
                 * @param imageAlreadySet
                 * @param data
                 * @param inspectionData
                 */
                fillPostsAndDrtPicture(step: any, filledPost: any, inspectionDataIndex: number, entryIndex: number,
                                       entry: any, imageAlreadySet: boolean, data: any, inspectionData: any): boolean {
                    let stepFound = false;
                    step.posts.some((post) => {
                        if (+post.id === filledPost.postId) {
                            let localImages = UtilsService.deepCopy(
                                this.inspectionData[inspectionDataIndex].steps[entryIndex].images
                            );
                            // Remove isFromThingworx images
                            localImages = localImages.filter(localImage => !localImage.isFromThingworx);
                            if (entry.mediaUuid && !imageAlreadySet) {
                                // Image handling
                                localImages.push({uuid: data.data[entryIndex].mediaUuid, isFromThingworx: true});
                                step.images = localImages;
                                imageAlreadySet = true;
                            }
                            post.value = filledPost.controlValue;
                            this.onRoutingComponentInspectionChanged(
                                {
                                    id: step.stepActivation.id,
                                    postThresholdId: post.thresholdId,
                                    value: parseFloat(post.value),
                                    status: inspectionData.status,
                                    idRoutingComponentIndex: inspectionData.id,
                                    images: step.images
                                }
                            );
                            stepFound = true;
                            return stepFound;
                        }
                        return stepFound;
                    });
                    return imageAlreadySet;
                }

                handlePosts(value: InspectionDataVisual, choice: any): InspectionDataVisual {

                                postThresholdId: choice.postThresholdId,

                            }
                        );
                    }
