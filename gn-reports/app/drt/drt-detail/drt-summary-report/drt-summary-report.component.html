                /**
                 * Get Summary report data
                 */
                getData(): void {
                    this.spinnerService.startLoading();
                    const urlGetSummaryReport = `/drts/${this.drtId}/summary-report`;
                    const paramUrl = this.isForReport ? '?language=en' : '';
                    this.drtSummaryReportGenese.getOneCustom(`${urlGetSummaryReport}${paramUrl}`)
                        .pipe(finalize(() => {
                            this.spinnerService.stopLoading();
                        }))
                        .subscribe((inspectionSummaryReport) => {
                            if (inspectionSummaryReport) {
                                this.inspectionSummaryReport = inspectionSummaryReport;
                                if (this.isForReport) {
                                    this.exportSummaryReport.emit();
                                }
                                if (this.inspectionSummaryReport && Array.isArray(this.inspectionSummaryReport.functionalAreaDamages)) {
                                    this.displayedInspectionSummaryReport = UtilsService.deepCopy(this.inspectionSummaryReport);
                                    this.setDamageCount();
                                    this.setDamagesList();
                                }
                            }
                    });
                }

                /**
                 * Set damages list
                 */
                setDamagesList(): void {
                    this.displayedInspectionSummaryReport.functionalAreaDamages.forEach(functionalAreaDamage => {
                        if (Array.isArray(functionalAreaDamage.damages)) {
                            for (const damage of functionalAreaDamage.damages) {
                                const mappedDamage: {id: string, name: string} = {id: damage.id.toString(), name: damage.name};
                                const index = this.damagesList.findIndex(damageElement => {
                                    return damageElement.id === mappedDamage.id;
                                });
            
                                if (index === -1) {
                                    this.damagesList.push(mappedDamage);
                                }
                            }
                        }
                    });
                }

                /**

                changeSelectedOption(selectedId: string): void {
                    this.damageSelected = selectedId;

                    if (this.inspectionSummaryReport) {

                                const tempInspectionSummaryReportDamages = UtilsService.deepCopy(this.inspectionSummaryReport.functionalAreaDamages);
                                this.displayedInspectionSummaryReport.functionalAreaDamages =

                                    tempInspectionSummaryReportDamages[functionalAreaDamageIndex].damages =
                                    functionalAreaDamage.damages.filter(x => x.id === selectedIdNumber);

                                }));
