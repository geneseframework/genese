                mapToDrtInspection(data: any): DrtInspection {
                    const inspection: DrtInspection = new DrtInspection();
                    inspection.damageName = data.damage ? UtilsService.default(data.damage.name, '') : '';
                    inspection.id = UtilsService.default(data.idRoutingComponentIndex, '');
                    inspection.status = UtilsService.default(data.statusOperation, Status.CREATED);
                    inspection.version = UtilsService.default(data.version, 1);
                    inspection.qcheckValue = data.qcheckValue !== undefined ? data.qcheckValue : undefined;
                    let countAuto = 1;
                    let countManu = 1;
                    if (data.steps && Array.isArray(data.steps) && data.steps.length > 0) {
                        const steps = [];
                        data.steps.forEach(s => {
                            const posts = [];
                            s.posts.forEach(p => {
                                const post = {
                                    designation: UtilsService.default(p.designation, ''),
                                    id: UtilsService.default(p.id, 0),
                                    measureUnit: p.measureUnit ? UtilsService.default(p.measureUnit.name, '') : '',
                                    threshold: p.postThreshold ? UtilsService.default(p.postThreshold.threshold, undefined) : undefined,
                                    thresholdId: p.postThreshold ? UtilsService.default(p.postThreshold.id, 0) : 0,
                                    value: p.value !== undefined ? p.value !== null ? UtilsService.default(p.value, false) : undefined : undefined
                                };
                                posts.push(post);
                            });
            
                            let nameConcatenated = '';
                            if (s.type === StepType.AUTO) {
                                this.subscriptions.push(this.translateService.get('APP.ROUTING.DETAIL.STEP_AUTOMATIC').subscribe(translate => {
                                    nameConcatenated = `${translate} ${data.numberStep ? data.numberStep : countAuto}${s.name ? ' - ' + s.name : ''}`;
                                    countAuto++;
                                }));
                            } else {
                                this.subscriptions.push(this.translateService.get('APP.ROUTING.DETAIL.STEP_MANUAL').subscribe(translate => {
                                    nameConcatenated = `${translate} ${data.numberStep ? data.numberStep : countManu}${s.name ? ' - ' + s.name : ''}`;
                                    countManu++;
                                }));
                            }
            
                            const step = {
                                files: UtilsService.default(s.files, []),
                                id: UtilsService.default(s.id, ''),
                                activated: s.stepActivation ? UtilsService.default(s.stepActivation.activated, undefined) : undefined,
                                activatedByOperator: s.stepActivation ?
                                    (s.stepActivation.drtPictures) ?
                                        UtilsService.default(s.stepActivation.drtPictures.activated, undefined)
                                        : undefined
                                    : undefined,
                                images: s.stepActivation ?
                                    (s.stepActivation.drtPictures) ?
                                        UtilsService.default(s.stepActivation.drtPictures.images, [])
                                        : []
                                    : [],
                                information: s.information ? UtilsService.default(s.information, '') : '',
                                name: s.name ? UtilsService.default(s.name, '') : '',
                                nameConcatenated,
                                value: UtilsService.default(data.todoListValue, ''),
                                posts,
                                stepNumber: UtilsService.default(+s.stepNumber, 0),
                                stepActivation: s.stepActivation ? UtilsService.deepCopy(s.stepActivation) : {},
                                type: UtilsService.default(s.type, undefined),
                            };
                            steps.push(step);
                        });
                        inspection.steps = steps;
                    }
                    return inspection;
                }
