                /**
                 * Set operations list status

                 * @param taskSelectedIndex
                 * @param operationIndex

                setOperationsListStatus(taskToHandle: any[], taskSelectedIndex: number, operationIndex: number, operation: DrtOperation, result: {
                    reason: string,
                    value: InspectionData,
                    qcheckFullyTrue?: boolean
                }): void {
                    if (taskToHandle[taskSelectedIndex].status === Status.CREATED) {
                        this.operationsStatus[operationIndex].notInCreatedStateInspectionsNumber++;
                    }
                    if (result.reason === 'validate') {
                        if (taskToHandle[taskSelectedIndex].status === Status.Q_CHECK) {
                            if (result.qcheckFullyTrue === true) {
                                taskToHandle[taskSelectedIndex].status = Status.VALIDATED;
                            } else {
                                taskToHandle[taskSelectedIndex].status = Status.IN_PROGRESS;
                            }
                        } else {
                            if (operation.behavior === 'TODO_LIST' || operation.behavior === 'VISUAL_COMPONENT') {
                                taskToHandle[taskSelectedIndex].status = Status.VALIDATED;
                            } else {
                                if (taskToHandle[taskSelectedIndex].status !== Status.VALIDATED) {
                                    taskToHandle[taskSelectedIndex].status = Status.Q_CHECK;
                                    this.operationsStatus[operationIndex].usefulStatus = Status.Q_CHECK;
                                }
                            }
                        }
                        if (taskToHandle.every(task => task.status === Status.VALIDATED)) {
                            this.operationsStatus[operationIndex].usefulStatus = Status.VALIDATED;
                        } else if (this.operationsStatus[operationIndex].usefulStatus !== Status.Q_CHECK) {
                            this.operationsStatus[operationIndex].usefulStatus = Status.IN_PROGRESS;
                        }
                        if (this.operationsStatus.every(status => {
                            return status.usefulStatus === Status.VALIDATED;
                        })) {
                            this.isClosable = true;
                        }
                    } else {
                        if (this.operationsStatus[operationIndex].usefulStatus !== Status.Q_CHECK) {
                            this.operationsStatus[operationIndex].usefulStatus = Status.IN_PROGRESS;

                                if (x.id) {
                                    return x.id.toString() === this.taskSelected.task;
                                } else {
                                    return false;
                                }
                            });
                            !!task ? task.status = Status.IN_PROGRESS : noop();
                        }
                    }
                }

                /**
                 * Set operations todolist
                 * @param operation
                 * @param notInCreatedStateInspectionsNumber
                 * @param usefulStatus
                 */
                setOperationsTodolist(operation: DrtOperation, notInCreatedStateInspectionsNumber: number, usefulStatus: Status):
                    {notInCreatedStateInspectionsNumber: number, usefulStatus: Status} {
                    operation.todoLists.forEach((status) => {
                        if (status.status !== Status.CREATED) {
                            notInCreatedStateInspectionsNumber++;
                        }
                        if (status.status === Status.IN_PROGRESS &&
                            (usefulStatus === Status.CREATED || usefulStatus === Status.IN_PROGRESS)) {
                            // Orange case
                            usefulStatus = status.status;
                        } else if (status.status !== Status.CREATED) {
                            if (usefulStatus === Status.VALIDATED && status.status !== Status.VALIDATED) {
                                usefulStatus = status.status;
                            } else if (usefulStatus === Status.IN_PROGRESS && status.status !== Status.IN_PROGRESS) {
                                noop();
                            } else {
                                // All other cases are green
                                usefulStatus = Status.VALIDATED;
                            }
                        }
                    });
                    return {notInCreatedStateInspectionsNumber, usefulStatus};
                }

                /**
                 * Set operations other than todolist
                 * @param operation
                 * @param notInCreatedStateInspectionsNumber
                 * @param usefulStatus
                 */
                setOperationsOtherThanTodolist(operation: DrtOperation, notInCreatedStateInspectionsNumber: number, usefulStatus: Status):
                    {notInCreatedStateInspectionsNumber: number, usefulStatus: Status} {
                    let functionalAreaValidated = 0;
                    operation.functionalAreaStatus.forEach((status) => {
                        if (status.status !== Status.CREATED) {
                            notInCreatedStateInspectionsNumber++;
                        }
                        if ((status.status === Status.IN_PROGRESS || status.status === Status.Q_CHECK) &&
                            (usefulStatus === Status.CREATED || usefulStatus === Status.IN_PROGRESS)) {
                            // Orange case
                            usefulStatus = status.status;
                        } else if (status.status !== Status.CREATED) {
                            functionalAreaValidated++;
                        }
                    });
                    if (functionalAreaValidated === operation.functionalAreaStatus.length) {
                        usefulStatus = Status.VALIDATED;
                    } else if (functionalAreaValidated > 0) {
                        usefulStatus = Status.IN_PROGRESS;
                    }
                    return {notInCreatedStateInspectionsNumber, usefulStatus};
                }
