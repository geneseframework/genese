                /**
                 * Save inspection
                 * @param validate
                 */
                saveInspection(validate: boolean): void {
                    const localInspectionData: InspectionData = UtilsService.deepCopy(this.inspectionData);
                    if (localInspectionData.todoListValue === '') { // ---------------------------------------------------------------- +1 Cognitive complexity (+1 break flow)
                        delete localInspectionData.todoListValue;
                    }
                    if (Array.isArray(localInspectionData.steps) && localInspectionData.steps.length > 0) { // ------------------------ +2 Cognitive complexity (+2 break flow)
                        localInspectionData.steps.forEach((step) => { // -------------------------------------------------------------- +2 Cognitive complexity (+1 break flow, +1 nesting)
                            if (Array.isArray(step.posts) && step.posts.length === 0) { // -------------------------------------------- +4 Cognitive complexity (+2 break flow, +2 nesting)
                                delete step.posts;
                            }
                            if (step.drtPictures && Array.isArray(step.drtPictures.images)) { // -------------------------------------- +4 Cognitive complexity (+2 break flow, +2 nesting)
                                step.drtPictures.images.forEach((image) => { // ------------------------------------------------------- +4 Cognitive complexity (+1 break flow, +3 nesting)
                                    delete image.isFromThingworx;
                                });
                            }
                        });
                    }
                    delete localInspectionData.status;
            
                    this.drtDataService.updateInspections(
                        this.drtId,
                        this.drtService.operationId,
                        this.drtService.taskId,
                        localInspectionData,
                        validate)
