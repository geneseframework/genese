                /**
                 * Transform string format from PascalCase to snake-case
                 * @param word
                 */
                /* istanbul ignore next */
                static toSnakeCase(word: string): string {
                    if (!word) { // ---------------------------------------------------- +1 Cognitive complexity (+1 break flow)
                        return '';
                    }
                    let snake = word.charAt(0).toLowerCase();
                    for (let i = 1; i < word.length; i++) { // ------------------------- +1 Cognitive complexity (+1 break flow)
                        if (word.charAt(i) === word.charAt(i).toUpperCase()) { // ------ +2 Cognitive complexity (+1 break flow, +1 nesting)
                            snake += '-' + word.charAt(i).toLowerCase();
                        } else {
                            snake += word.charAt(i); // -------------------------------- +1 Cognitive complexity (+1 break flow)
                        }
                    }
                    return snake;
                }

                static deepClone(model: any): any {
                    if (model) { // --------------------------------------------------------------------------------- +1 Cognitive complexity (+1 break flow)
                        if (Array.isArray(model)) { // -------------------------------------------------------------- +2 Cognitive complexity (+1 break flow, +1 nesting)
                            const newArray = [];
                            model.forEach(item => newArray.push(this.deepClone(item))); // -------------------------- +3 Cognitive complexity (+1 break flow, +2 nesting)
                            return newArray;
                        } else if (typeof model === 'object') { // -------------------------------------------------- +4 Cognitive complexity (+2 break flow, +2 nesting)
                            const newObject = {};
                            Object.keys(model).forEach(key => newObject[key] = this.deepClone(model[key])); // ------ +4 Cognitive complexity (+1 break flow, +3 nesting)
                            return newObject;
                        } else {
                            return model; // ------------------------------------------------------------------------ +1 Cognitive complexity (+1 break flow)
                        }
                    } else {
                        return model; // ---------------------------------------------------------------------------- +1 Cognitive complexity (+1 break flow)
                    }
                }
