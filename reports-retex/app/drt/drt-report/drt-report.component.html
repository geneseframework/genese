                getDetail(name: string, operationId: string, taskId: string, index: number, subIndex: number): void {
                    if (operationId && taskId) { // -------------------------------------------------------------------------------- +2 Cognitive complexity (+2 break flow)
                        this.spinnerService.startLoading();
                        const urlOperationDetail = `/drts/${this.drtId}/operations/${operationId}/tasks/${taskId}`;
            
                        this.inspectionDetailGenese.getAllCustom(`${urlOperationDetail}?language=en`)
                            .pipe(finalize(() => this.spinnerService.stopLoading())) // -------------------------------------------- +2 Cognitive complexity (+1 break flow, +1 nesting)
                            .subscribe((inspections: InspectionGenese[]) => { // --------------------------------------------------- +2 Cognitive complexity (+1 break flow, +1 nesting)
                                if (Array.isArray(inspections) && inspections[0]) { // --------------------------------------------- +4 Cognitive complexity (+2 break flow, +2 nesting)
                                    if (!this.operationTemp[index]) { // ----------------------------------------------------------- +4 Cognitive complexity (+1 break flow, +3 nesting)
                                        this.operationTemp[index] = [];
                                    }
                                    this.operationTemp[index][subIndex] = [];
                                    if (inspections.length > 0 && inspections[0].behavior === 'VISUAL_COMPONENT') { // ------------- +5 Cognitive complexity (+2 break flow, +3 nesting)
                                        inspections.forEach((inspection, indexStep) => { // ---------------------------------------- +5 Cognitive complexity (+1 break flow, +4 nesting)
                                            this.operationTemp[index][subIndex][indexStep] = {
                                                type: name, operationId, taskId, inspection
                                            };
                                        });
                                    } else if (Array.isArray(inspections[0].steps)) { // ------------------------------------------- +6 Cognitive complexity (+2 break flow, +4 nesting)
                                        inspections[0].steps.forEach((step, indexStep) => { // ------------------------------------- +6 Cognitive complexity (+1 break flow, +5 nesting)
                                            // We create one inspection by step to make work the page break of the report.
                                            const inspection = UtilsService.deepCopy(inspections[0]);
                                            inspection.steps = [UtilsService.deepCopy(step)];
                                            inspection.numberStep = indexStep + 1;
                                            this.operationTemp[index][subIndex][indexStep] = {
                                                type: name, operationId, taskId, inspection
                                            };
                                        });
                                    }
                                    this.numInspectionHandled++;
                                    if (this.numInspection === this.numInspectionHandled) { // ------------------------------------- +4 Cognitive complexity (+1 break flow, +3 nesting)
                                        // This is made to keep the order of the inspections in the DRT
                                        this.operationTemp.forEach(operation => { // ----------------------------------------------- +5 Cognitive complexity (+1 break flow, +4 nesting)

                                                    }
                                                });
                                            } else {
                                                this.operations.push(operation); // ------------------------------------------------ +1 Cognitive complexity (+1 break flow)
                                            }
                                        });

                }
