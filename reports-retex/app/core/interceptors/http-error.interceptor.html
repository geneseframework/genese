                /**
                 * Intercept Error HTTP from a request and redirect to Error page.
                 * @param {HttpRequest<any>} request - Request send to server
                 * @param {HttpHandler} next - Handle HTTP request
                 * @returns {Observable}
                 */

                    return next.handle(request.clone()).pipe(
