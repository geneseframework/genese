                static mapToStep(data: any): Step {
            
                    const attachedFiles: { uuid: string, filename: string, extension: string }[] = [];
                    if (data && Array.isArray(data.files)) { // --------------------------------------------------------------------------- +2 Cognitive complexity (+2 break flow)
                        data.files.forEach((file) => { // --------------------------------------------------------------------------------- +2 Cognitive complexity (+1 break flow, +1 nesting)
                            let extension = '';
                            if (file.filename) { // --------------------------------------------------------------------------------------- +3 Cognitive complexity (+1 break flow, +2 nesting)
                                const fileNameSplit = file.filename.split('.');
                                extension = fileNameSplit.pop();
                            }
                            attachedFiles.push({uuid: file.uuid, filename: file.filename, extension});
                        });
                    }
            
                    let translatedFields = {};
                    if (data && data.translatedFields && Object.keys(data.translatedFields).length) { // ---------------------------------- +2 Cognitive complexity (+2 break flow)
                        for (const key of Object.keys(data.translatedFields)) { // -------------------------------------------------------- +2 Cognitive complexity (+1 break flow, +1 nesting)
                            translatedFields[key] = {

                            };

                        translatedFields = {

                            },
                            FR: {
                                name: '',
                                information: ''
                            }
                        };
